from flask import Flask, render_template_string

app = Flask(__name__)


@app.route('/layout')
def layout():
    title = "Layout Demo"

    template = """
    {% extends 'base.html' %}
    {% set subtitle = "Reusable" %}
    {% block content %}
    <h2>{{ title }}</h2>
    <p>{{ subtitle }}</p>
    {% include 'partial.html' %}
    {% endblock %}
    """

    return render_template_string(template, title=title)


if __name__ == "__main__":
    app.run(debug=True)
